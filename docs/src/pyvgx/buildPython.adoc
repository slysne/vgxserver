== Build


=== MacOS

[source, bash]
----
# Make compatible from 14 and up
export MACOSX_DEPLOYMENT_TARGET=14.0
DIR_NAME=$(basename "$PWD")
PY_MAJOR=$(echo "$DIR_NAME" | sed -E 's/^Python-[0-9]+\.([0-9]+)\..*/\1/')

# Set up LDFLAGS for rpath â€” helpful for embedded or relocatable binaries
export LDFLAGS="-Wl,-rpath,@loader_path"

./configure \
    --prefix=$HOME/python3$PY_MAJOR \
    --enable-shared \
    --enable-optimizations \
    --with-openssl=$(brew --prefix openssl)

make -j$(sysctl -n hw.ncpu)

make install

install_name_tool -id @rpath/libpython3.$PY_MAJOR.dylib \
    $HOME/python3$PY_MAJOR/lib/libpython3.$PY_MAJOR.dylib

cd ~

python3$PY_MAJOR/bin/python3 -m venv venv3$PY_MAJOR

. venv3$PY_MAJOR/bin/activate

pip install --upgrade pip

pip install --upgrade setuptools build wheel delocate


----
